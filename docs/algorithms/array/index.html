Массивы

Массивы - наиболее часто встречающаяся структура данных, с которой приятно работать. Основные задачи связанные с массивами - это задачи поиска и сортировки. Рассмотрим же некоторые из них!

Одномерные статические массивы​

Задачи, в которых используются одномерные статически массивы - наиболее простые задачи, которые могут попасться вам под руку. Это обусловленно тем, что со статическим массивом обычно меньше всего проблем: он не использует динамическую память, его размер определен заранее, а значит операции над ним нас ждут весьма тривиальные.

Задача: Сортировать массив известными методами​

Сортировка пузырьком! думаю, что тут даже не требуется дополнительного описания. Но на самом деле она звучит так

Отсортируйте по неубыванию методом "пузырька" одномерный целочисленный массив, заданный случайными числами на промежутке [-100; 100). Выведите на экран исходный и отсортированный массивы.

Решение
#include "bubble.h"

//функция для сортировки методом пузырька
void SortBubble (int arr[], int size) {
    for (int i = 0; i < size; ++i){
        for (int j = 0; j < size; ++j){
            if (arr[j] > arr [j+1]){
                int tmp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = tmp;
            }
        }
    }
}

//вывод массива в стандартный поток вывода
void PrintBubble(int arr[], int size) {
    for (int i = 0; i < size; ++i) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
}

void ExecuteBubbleSort (){
    //инициализируем генератор случайных чисел
    srand(static_cast<unsigned int>(time(0)));

    //ввод размера массива
    std::cout << "Enter the size of the array: " << std::endl;
    int size;
    std::cin >> size;

    //будем использовать умный указатель, для создания массива в
    //динамической паямти
    std::unique_ptr<int[]> arr(new int[size]);

    //заполняем массив в диапозоне от -100 до 99
    for (int i = 0; i < size; ++i) {
        arr[i] = rand() % 200 - 100;
    }

    //вызов функции печати массива после заполнения
    std::cout << "Original array: " << std::endl;
    PrintBubble(arr.get(), size);

    //вызов функции сортировки
    SortBubble(arr.get(), size);

    //вызов функции печати массива после сортировки
    std::cout << "Sorted array: " << std::endl;
    PrintBubble(arr.get(), size);

    // указатель освобождается при выходе из области видимости.
    arr.reset(); // принудительно освобождаем память

    // проверка, что указатель стал nullptr после освобождения памяти
    assert(arr.get() == nullptr);
    std::cout << "Memory successfully freed." << std::endl;

}

Задача 2

Отсортируйте по невозрастанию методом простого выбора одномерный вещественный массив, заданный случайными числами на промежутке [0; 50). Выведите на экран исходный и отсортированный массивы.

Решение
#include "selection.h"

//функция для сортировки
void SelectionSort (double arr[], int size) {
    for (int i = 0; i < size; ++i){
        int maxIndex;
        for (int j = 0; j <size; ++j){
            if (arr[j] > maxIndex){
                maxIndex = j;
            }
        }
        double tmp = arr[i];
        arr[i] = arr[maxIndex];
        arr[maxIndex] = tmp;
    }
}

//вывод массива в стандартный поток вывода
void PrintSelection(double arr[], int size) {
    for (int i = 0; i < size; ++i) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
}

void ExecuteSelectionSort (){
    //инициализируем генератор случайных чисел
    srand(static_cast<unsigned int>(time(0)));

    //ввод размера массива
    std::cout << "Enter the size of the array: " << std::endl;
    int size;
    std::cin >> size;

    //будем использовать умный указатель, для создания массива в
    //динамической паямти
    std::unique_ptr<double[]> arr(new double [size]);

    //заполняем массив в диапозоне [0; 50)
    for (int i = 0; i < size; ++i) {
        arr[i] = static_cast<double>(rand()) / RAND_MAX * 50;
    }

    //вызов функции печати массива после заполнения
    std::cout << "Original array: " << std::endl;
    PrintSelection(arr.get(), size);

    //вызов функции сортировки
    SelectionSort(arr.get(), size);

    //вызов функции печати массива после сортировки
    std::cout << "Sorted array: " << std::endl;
    PrintSelection(arr.get(), size);

    // указатель освобождается при выходе из области видимости.
    arr.reset(); // принудительно освобождаем память

    // проверка, что указатель стал nullptr после освобождения памяти
    assert(arr.get() == nullptr);
    std::cout << "Memory successfully freed." << std::endl;

}

Задача 3

Отсортируйте по возрастанию методом простого включения одномерный целочисленный массив, заданный с клавиатуры различными числами. Выведите на экран исходный и отсортированный массивы.

Решение
#include "insertion.h"

void insertionSort(int arr[], int size){
    for (int i = 0; i < size; ++i){
        int key = arr[i];
        int j = i - 1;

        while (j >= 0 && arr[j] > key){
            arr[j + 1] = arr[j];
            --j;
        }
        arr[j+1] = key;
    }
}

void PrintInsertion(int arr[], int size) {
    for (int i = 0; i < size; ++i) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
}

// Основная функция для выполнения сортировки включением
void ExecuteInsertionSort() {
    int size;
    std::cout << "Enter the size of the array: ";
    std::cin >> size;

    std::unique_ptr<int[]> arr(new int[size]);

    std::cout << "Fill the array with distinct numbers: ";
    std::cout << " ";
    for (int i = 0; i < size; ++i) {
        int input;
        bool isDuplicate;

        do {
            isDuplicate = false;
            std::cin >> input;

            for (int j = 0; j < i; ++j){
                if (arr[j] == input){
                    std::cout << "Number already entered. Please try another: ";
                    isDuplicate = true;
                    break;
                }
            }
        } while (isDuplicate);
        arr[i] = input;
    }

    std::cout << "Original array: ";
    PrintInsertion(arr.get(), size);

    insertionSort(arr.get(), size);

    std::cout << "Sorted array: ";
    PrintInsertion(arr.get(), size);

    arr.reset();

    assert(arr.get() == nullptr);
    std::cout << "Memory successfully freed." << std::endl;
}

Задача 4

Массив размером 2m + 1, где m – натуральное число, заполнен случайным образом. Найдите в массиве медиану. Медианой называется элемент ряда, делящий его на две равные части: в одной находятся элементы, которые не меньше медианы, в другой – не больше медианы.

Решение
#include <iostream>
#include <algorithm>
#include <memory>

#pragma once

void FindMedian () {
    int m;
    std::cout << "Enter the value of m (natural number): ";
    std::cin >> m;

    if (m <= 0){
        std::cout << "The value of m must be a natural number (positive integer)." << std::endl;
        return;
    }

    int size = 2*m + 1;

    std::unique_ptr<int[]> arr(new int[size]);

    //заполняем массив в диапозоне [0; 50)
    for (int i = 0; i < size; ++i) {
        arr[i] = static_cast<double>(rand()) / RAND_MAX * 50;
    }

    std::cout << "Generated array: ";

    for (int i = 0; i < size; ++i ){
        std::cout << arr[i] << ' ';
    }

    std::cout << std::endl;

    // Сортировка массива
    std::sort(arr.get(), arr.get()+size);

    //В задании вроде выводить его не надо, но включил для наглядности, все равно сортирую его
    std::cout << "Sorted array: ";

    for (int i = 0; i < size; ++i ){
        std::cout << arr[i] << ' ';
    }

    // Нахождение медианы
    int median = arr[size/2];
    std::cout << "Median: " << median << std::endl;

    arr.reset();

    assert(arr.get() == nullptr);
    std::cout << "Memory successfully freed." << std::endl;

}

Задача 5

Массив размером m, где m – натуральное число, заполнен случайным образом. Найдите в массиве моду. Модой называется элемент ряда, который встречается наиболее часто.

Решение
#include <iostream>
#include <unordered_map>
#include <algorithm>
#include <vector>
#include <memory>

#pragma once

void FindMode () {
    int m;
    std::cout << "Enter the value of m (size) (natural number): ";
    std::cin >> m;

    if (m <= 0){
        std::cout << "The value of m must be a natural number (positive integer)." << std::endl;
        return;
    }

    std::unique_ptr<int[]> arr(new int[m]);

    //заполняем массив в диапозоне [0; 50)
    for (int i = 0; i < m; ++i) {
        arr[i] = static_cast<double>(rand()) / RAND_MAX * 50;
    }

    std::cout << "Generated array: ";

    for (int i = 0; i < m; ++i ){
        std::cout << arr[i] << ' ';
    }

    std::cout << std::endl;

    // Подсчет вхождений элементов
    // Будем использовать unordered_map для подсчета вхождения чисел
    std::unordered_map<int, int> frequencyMap;
    for (int i = 0; i < m; ++i) {
        frequencyMap[arr[i]]++;
    }

    // Определение максимальной частоты
    int maxFrequency = 0;
    for (const auto& pair : frequencyMap) {
        if (pair.second > maxFrequency) {
            maxFrequency = pair.second;
        }
    }

    // Проверка крайних случаев, все ли числа уникальны
    if (maxFrequency == 1) {
        std::cout << "All numbers in the array occur exactly once." << std::endl;
    } else {
        std::cout << "Numbers with the maximum frequency (" << maxFrequency << "):" << std::endl;
        for (const auto& pair : frequencyMap) {
            if (pair.second == maxFrequency) {
                std::cout << "Number: " << pair.first << std::endl;
            }
        }
    }

    arr.reset();

    assert(arr.get() == nullptr);
    std::cout << "Memory successfully freed." << std::endl;
}

Задача: Обновление состава дроидов (поиск двух минимумов)​

В связи с визитом Императора Палпатина было решено обновить состав дроидов в ангаре 32. Из-за кризиса было принято решение не закупать новых дроидов, а просто убрать несколько старых. Как известно, Палпатин не переносит дроидов с маленькими серийными номерами, поэтому задача состоит в том, чтобы найти среди них двух дроидов с наименьшими серийными номерами.

Формат входного файла​

  • Первая строка входного файла содержит целое число N — количество дроидов (2 ≤ N ≤ 1000).
  • Вторая строка содержит N целых чисел, по модулю не превышающих 2 × 10^9 — серийные номера дроидов.
Решение
#include <iostream>
#include <climits>  // Для INT_MAX

int main() {
    int n;
    // Считываем количество элементов массива
    std::cin >> n;

    // Создаем массив для хранения результатов
    int arr[n];
    // Определяем размер массива (здесь размер фиксирован и соответствует n)
    int size = sizeof(arr) / sizeof(arr[0]);

    // Считываем элементы массива
    for (int i = 0; i < size; ++i){
        int value;
        std::cin >> value;
        arr[i] = value;  // Заполняем массив введенными значениями
    }

    // Если количество элементов меньше или равно 2, невозможно найти два различных минимальных элемента
    if (n <= 2) {
        std::cout << "Destroy all droids!" << std::endl;
        return -1;  // Завершаем программу с кодом ошибки
    }

    // Инициализируем переменные для хранения минимальных значений
    int min1 = INT_MAX;  // Минимальное значение
    int min2 = INT_MAX;  // Второе минимальное значение

    // Поиск двух минимальных значений в массиве
    for (int i = 0; i < size; ++i){
        if (arr[i] < min1) {
            // Если текущее значение меньше текущего минимального, обновляем min1 и min2
            min2 = min1;
            min1 = arr[i];
        } else if (arr[i] < min2 && arr[i] != min1) {
            // Если текущее значение меньше min2, но больше min1 и не равно min1
            min2 = arr[i];
        }
    }

    // Если min2 осталась равной INT_MAX, это значит, что все значения в массиве равны
    if (min2 == INT_MAX) {
        std::cout << "All droids have equal numbers" << std::endl;
    } else {
        // Выводим два минимальных значения
        std::cout << min1 << ' ' << min2 << std::endl;
    }

    return 0;  // Успешное завершение программы
}


Все эти реализации работают с массивами, которые создаются уже заданного размера, поэтому алгоритмы работы с ними достаточно простые.

Динамические массивы​

Задачи с динамическими массивами - обычно задачи более сложные. Они обычно начинаются со слов: "Задан массива размера N". Такие задачи чаще направлены на поиск алгоритма, используют в своих решениях указатели и более сложные алгоритмы решения.

Задача: бинарный поиск​

Некто загадал число от 1 до N. За какое наименьшее количество вопросов (на которые он отвечает "да" или "нет") можно угадать задуманное число?

Да, это не совсем про массивы, но бинарный поиск все таки отсюда.

Решение
#include <iostream>
#include <cmath>

int main() {
    int n;
    std::cin >> n;
    //элегантное решение через логорифм
    int qustions = std::ceil(std::log2(n));

    std::cout << qustions << std::endl;

    return 0;
}

Задача: Седловые числа​

Задана матрица K, содержащая n строк и m столбцов. Седловой точкой этой матрицы назовем элемент, который одновременно является минимумом в своей строке и максимумом в своем столбце. Найдите количество седловых точек заданной матрицы.

Решение
#include <iostream>
#include <vector>
#include <limits.h>  // Для INT_MAX, если потребуется

int main() {
    int n, m;
    // Считываем количество строк (n) и столбцов (m) матрицы
    std::cin >> n >> m;

    // Создаем матрицу размером n x m и инициализируем её нулями
    std::vector<std::vector<int>> matrix(n, std::vector<int>(m));

    // Ввод элементов матрицы
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            std::cin >> matrix[i][j];  // Читаем значение и записываем в матрицу
        }
    }

    int saddlePointsCount = 0;  // Счетчик седловых точек

    // Поиск седловых точек
    for (int i = 0; i < n; ++i) {
        // Находим минимальное значение в строке i
        int rowMin = matrix[i][0];
        for (int j = 1; j < m; ++j) {
            if (matrix[i][j] < rowMin) {
                rowMin = matrix[i][j];  // Обновляем минимальное значение строки
            }
        }

        // Проверяем каждый элемент строки на выполнение условий седловой точки
        for (int j = 0; j < m; ++j) {
            if (matrix[i][j] == rowMin) {  // Если текущий элемент равен минимальному значению строки
                bool isSaddlePoint = true;

                // Проверяем, является ли этот элемент максимумом в своем столбце
                for (int k = 0; k < n; ++k) {
                    if (matrix[k][j] > matrix[i][j]) {
                        isSaddlePoint = false;  // Элемент не является максимумом в столбце
                        break;  // Выходим из цикла проверки столбца
                    }
                }

                // Если элемент удовлетворяет обоим условиям, увеличиваем счетчик седловых точек
                if (isSaddlePoint) {
                    saddlePointsCount++;
                }
            }
        }
    }

    // Выводим количество седловых точек
    std::cout << saddlePointsCount << std::endl;

    return 0;  // Завершение программы
}
